    <script>

        $(document).ready(function() {

            let scannedBarcodes = [];

            const modal = document.getElementById("options-modal");

            const closeBtn = document.getElementsByClassName("close")[0];

            const cameraSupported = 'mediaDevices' in navigator && 'getUserMedia' in navigator.mediaDevices;

            

            // Show camera option if supported

            if (cameraSupported) {

                $("#camera-option").show();

            }

            

            // Open modal when add button is clicked

            $('#add-button').on('click', function() {

                modal.style.display = "block";

            });

            

            // Close modal when X is clicked

            closeBtn.onclick = function() {

                modal.style.display = "none";

            };

            

            // Close modal when clicking outside of it

            window.onclick = function(event) {

                if (event.target == modal) {

                    modal.style.display = "none";

                }

            };

            

            // Manual entry option

            $('#manual-entry').on('click', function() {

                modal.style.display = "none";

                const barcodeInput = prompt("Enter barcode:", "");

                if (barcodeInput && barcodeInput.trim() !== "") {

                    addBarcodeToList(barcodeInput.trim());

                }

            });

            

            // Camera entry option (if available)

            $('#camera-entry').on('click', function() {

                modal.style.display = "none";

                

                // Try to access camera

                navigator.mediaDevices.getUserMedia({ video: { facingMode: "environment" } })

                    .then(function(stream) {

                        // Create a video element to show the camera feed

                        const videoModal = $('<div class="modal" style="display:block;">' +

                            '<div class="modal-content" style="padding:0; width:90%; max-width:600px;">' +

                            '<div style="padding:10px; display:flex; justify-content:space-between; align-items:center; background:#f8f9fa; border-radius:10px 10px 0 0;">' +

                            '<h3>Scan Barcode</h3>' +

                            '<span class="close" id="close-camera">&times;</span>' +

                            '</div>' +

                            '<video id="camera-preview" autoplay playsinline style="width:100%; height:300px; object-fit:cover;"></video>' +

                            '<div style="padding:15px; text-align:center;">' +

                            '<button id="capture-barcode" class="add-button" style="margin:0;">Capture</button>' +

                            '<p style="margin-top:10px; font-size:14px; color:#6c757d;">Position the barcode in the center of the screen</p>' +

                            '</div>' +

                            '</div>' +

                            '</div>');

                        

                        $('body').append(videoModal);

                        

                        const video = document.getElementById('camera-preview');

                        video.srcObject = stream;

                        

                        // Close camera modal

                        $('#close-camera').on('click', function() {

                            stream.getTracks().forEach(track => track.stop());

                            videoModal.remove();

                        });

                        

                        // Capture button - in a real app, this would analyze the video frame for barcodes

                        // Here we'll just simulate it with a prompt

                        $('#capture-barcode').on('click', function() {

                            stream.getTracks().forEach(track => track.stop());

                            videoModal.remove();

                            

                            // Simulate barcode reading with prompt

                            const barcodeInput = prompt("Enter the barcode you see (simulating camera capture):", "");

                            if (barcodeInput && barcodeInput.trim() !== "") {

                                addBarcodeToList(barcodeInput.trim());

                            }

                        });

                    })

                    .catch(function(err) {

                        console.error("Camera error:", err);

                        alert("Could not access camera. Please ensure camera permissions are granted and try again.");

                        

                        // Fall back to manual entry

                        const barcodeInput = prompt("Enter barcode manually:", "");

                        if (barcodeInput && barcodeInput.trim() !== "") {

                            addBarcodeToList(barcodeInput.trim());

                        }

                    });

            });

            

            // Generate random barcode option

            $('#generate-random').on('click', function() {

                modal.style.display = "none";

                

                // Generate a random EAN-13 barcode

                let randomBarcode = "";

                for (let i = 0; i < 13; i++) {

                    randomBarcode += Math.floor(Math.random() * 10);

                }

                

                addBarcodeToList(randomBarcode);

            });

            

            // Handle input field enter key

            $('#barcode-input').on('keyup', function(e) {

                if (e.key === 'Enter') {

                    const barcode = $(this).val().trim();

                    if (barcode !== "") {

                        addBarcodeToList(barcode);

                        $(this).val('');

                    }

                }

            });

            

            // Function to add barcode to list

            function addBarcodeToList(barcode) {

                if (!scannedBarcodes.some(item => item.barcode === barcode)) {

                    scannedBarcodes.push({

                        barcode: barcode,

                        brand: '',

                        manufactureDate: '',

                        type: '',

                        serialNumber: ''

                    });

                    

                    updateBarcodeTable();

                }

            }

            

            // Update the barcode table

            function updateBarcodeTable() {

                var tableBody = $('#barcode-list');

                tableBody.empty();

                

                if (scannedBarcodes.length === 0) {

                    tableBody.append('<tr class="no-items"><td colspan="5">No barcodes added yet</td></tr>');

                    return;

                }

                

                $.each(scannedBarcodes, function(index, item) {

                    var row = $('<tr>');

                    row.append($('<td>').text(item.barcode));

                    row.append($('<td>').text(item.brand));

                    row.append($('<td>').text(item.manufactureDate));

                    row.append($('<td>').text(item.type));

                    row.append($('<td>').text(item.serialNumber));

                    tableBody.append(row);

                });

            }

        });

    </script>
